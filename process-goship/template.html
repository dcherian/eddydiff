def compare_micro_fine_section(var, micromeans, finemeans):
    micromeans[var].attrs["long_name"] = finemeans[var].attrs["long_name"]

    fg = micromeans.sel(cleaner="GE")[var].cf.plot(
        y="pressure",
        col="station_bins",
        hue="direction",
        xscale="log",
        aspect=0.4,
        size=5,
        add_legend=False,
        xlim=(1e-12, 1e-5),
    )
    for idx, (name, ax) in enumerate(zip(fg.name_dicts.ravel(), fg.axes.ravel())):
        hdl = (
            micromeans[var]
            .sel(cleaner="none")
            .sel(name)
            .cf.plot.line(
                y="pressure",
                hue="direction",
                add_legend=False,
                ax=ax,
                _labels=False,
                lw=0.5,
            )
        )
        fg._mappables[idx].extend(hdl)

    for idx, (name, ax) in enumerate(zip(fg.name_dicts.ravel(), fg.axes.ravel())):
        hdl = (
            finemeans[var]
            .sel(name)
            .cf.plot.line(
                y="pressure",
                hue="criteria",
                add_legend=False,
                ax=ax,
                _labels=False,
                lw=1,
            )
        )
        fg._mappables[idx].extend(hdl)
        ax.grid(True, which="both")

    fg.fig.legend(
        handles=fg._mappables[-1],
        labels=[f"χpod GE {direction}" for direction in micromeans.direction.data]
        + [f"χpod {direction}" for direction in micromeans.direction.data]
        + [f"Fine {criteria}" for criteria in finemeans.criteria.data],
        bbox_to_anchor=(1.12, 0.8),
    )
    return fg
